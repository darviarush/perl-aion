494             =head1 DESCRIPTION
495              
496             Aion is OOP-framework for creating classes with B<features>, has B<aspects>, B<roles> and so on.
497              
498             The properties declared through HAS are called B<features>.
499              
500             And C<is>,C<isa>, C<default>, and so on inC<has> are called B<aspects>.
501              
502             In addition to standard aspects, roles can add their own aspects using the B<aspect> subprogram.
503              
504             The signature of the methods can be checked using the attribute C<:Isa(...)>.
505

774

803             =head1 METHODS
804              
805             =head2 has ($feature)
806              
807             Checks that the property is established.
808              
809             Features having C<< default =E<gt> sub {...} >> perform C<sub> during the first call of the Getter, that is: are delayed.
810              
811             C<< $object-E<gt>has('feature') >> allows you to check that C<default> has not yet been called.
812              
813             package ExHas { use Aion;
814             has x => (is => 'rw');
815             }
816            
817             my $ex = ExHas->new;
818            
819             $ex->has("x") # -> ""
820            
821             $ex->x(10);
822            
823             $ex->has("x") # -> 1
824              
825             =head2 clear (@features)
826              
827             He removes the keys of the feature from the object by previously calling them C<clearer> (if exists).
828              
829             package ExClearer { use Aion;
830             has x => (is => 'rw');
831             has y => (is => 'rw');
832             }
833            
834             my $c = ExClearer->new(x => 10, y => 12);
835            
836             $c->has("x") # -> 1
837             $c->has("y") # -> 1
838            
839             $c->clear(qw/x y/);
840            
841             $c->has("x") # -> ""
842             $c->has("y") # -> ""
843              
844             =head1 METHODS IN CLASSES
845              
846             C<Use Aion> includes the following methods in the module:
847              
848             =head2 new (%parameters)
849              
850             The constructor.
851              
852             =head1 ASPECTS
853              
854             C<use Aion> includes the following aspects in the module for use in C<has>:
855              
856             =head2 is => $permissions
857              
858             =over
859              
860             =item * C<ro> - create only a gutter.
861              
862             =item * C<wo> - create only a setter.
863              
864             =item * C<rw> - Create getter and setter.
865              
866             =back
867              
868             By default - C<rw>.
869              
870             Additional permits:
871              
872             =over
873              
874             =item * C<+> - feature is required in the parameters of the designer. C<+> is not used with C<->.
875              
876             =item * C<-> - a feature cannot be installed through the constructor. '-' is not used with C<+>.
877              
878             =item * C<*> - do not increase the counter of links to the value (apply C<weaken> to the value after installing it in a feature).
879              
880             =back
881              
882             package ExIs { use Aion;
883             has rw => (is => 'rw');
884             has ro => (is => 'ro+');
885             has wo => (is => 'wo-');
886             }
887            
888             eval { ExIs->new }; $@ # ~> \* Feature ro is required!
889             eval { ExIs->new(ro => 10, wo => -10) }; $@ # ~> \* Feature wo cannot set in new!
890             ExIs->new(ro => 10);
891             ExIs->new(ro => 10, rw => 20);
892            
893             ExIs->new(ro => 10)->ro # -> 10
894            
895             ExIs->new(ro => 10)->wo(30)->has("wo") # -> 1
896             eval { ExIs->new(ro => 10)->wo }; $@ # ~> has: wo is wo- \(not get\)
897             ExIs->new(ro => 10)->rw(30)->rw # -> 30
898              
899             The function with C<*> does not hold the meaning:
900              
901             package Node { use Aion;
902             has parent => (is => "rw*", isa => Maybe[Object["Node"]]);
903             }
904            
905             my $root = Node->new;
906             my $node = Node->new(parent => $root);
907            
908             $node->parent->parent # -> undef
909             undef $root;
910             $node->parent # -> undef
911            
912             # And by setter:
913             $node->parent($root = Node->new);
914            
915             $node->parent->parent # -> undef
916             undef $root;
917             $node->parent # -> undef
918              
919             =head2 isa => $type
920              
921             Indicates the type, or rather - a validator, feature.
922              
923             package ExIsa { use Aion;
924             has x => (is => 'ro', isa => Int);
925             }
926            
927             eval { ExIsa->new(x => 'str') }; $@ # ~> \* Feature x must have the type Int. The it is 'str'
928             eval { ExIsa->new->x }; $@ # ~> Get feature `x` must have the type Int. The it is undef
929             ExIsa->new(x => 10)->x # -> 10
930              
931             For a list of validators, see L<Aion:::Type>.
932              
933             =head2 default => $value
934              
935             The default value is set in the designer if there is no parameter with the name of the feature.
936              
937             package ExDefault { use Aion;
938             has x => (is => 'ro', default => 10);
939             }
940            
941             ExDefault->new->x # -> 10
942             ExDefault->new(x => 20)->x # -> 20
943              
944             If C<$ Value> is a subprogram, then the subprogram is considered a designer of the meaning of the feature. Lazy calculation is used.
945              
946             my $count = 10;
947            
948             package ExLazy { use Aion;
949             has x => (default => sub {
950             my ($self) = @_;
951             ++$count
952             });
953             }
954            
955             my $ex = ExLazy->new;
956             $count # -> 10
957             $ex->x # -> 11
958             $count # -> 11
959             $ex->x # -> 11
960             $count # -> 11
961              
962             =head2 trigger => $sub
963              
964             C<$sub> is called after installing the property in the constructor (C<new>) or through the setter.
965             Etymology - let in.
966              
967             package ExTrigger { use Aion;
968             has x => (trigger => sub {
969             my ($self, $old_value) = @_;
970             $self->y($old_value + $self->x);
971             });
972            
973             has y => ();
974             }
975            
976             my $ex = ExTrigger->new(x => 10);
977             $ex->y # -> 10
978             $ex->x(20);
979             $ex->y # -> 30
980              
981             =head2 release => $sub
982              
983             C<$sub> is called before returning the property from the object through the gutter.
984             Etymology - release.
985              
986             package ExRelease { use Aion;
987             has x => (release => sub {
988             my ($self, $value) = @_;
989             $_[1] = $value + 1;
990             });
991             }
992            
993             my $ex = ExRelease->new(x => 10);
994             $ex->x # -> 11
995              
996             =head2 clearer => $sub
997              
998             C<$sub> is called when the deructor is called orC<< $object-E<gt>clear("feature") ``, but only if there is a property (see >>$object->has(" feature ")`).
999              
1000             package ExClearer { use Aion;
1001            
1002             our $x;
1003            
1004             has x => (clearer => sub {
1005             my ($self) = @_;
1006             $x = $self->x
1007             });
1008             }
1009            
1010             $ExClearer::x # -> undef
1011             ExClearer->new(x => 10);
1012             $ExClearer::x # -> 10
1013            
1014             my $ex = ExClearer->new(x => 12);
1015            
1016             $ExClearer::x # -> 10
1017             $ex->clear('x');
1018             $ExClearer::x # -> 12
1019            
1020             undef $ex;
1021            
1022             $ExClearer::x # -> 12
