433             If subroutine make'nt, then died.

437             =head2 make_maybe_arg ($pkg)
438              
439             It make subroutine with or without arguments, who return type.
440              
441             BEGIN {
442             Aion::Type->new(
443             name => "Enum123",
444             test => sub { $_ ~~ [1,2,3] },
445             a_test => sub { $_ ~~ $Aion::Type::SELF->{args} },
446             )->make_maybe_arg(__PACKAGE__);
447             }
448            
449             3 ~~ Enum123 # -> 1
450             3 ~~ Enum123[4,5,6] # -> ""
451             5 ~~ Enum123[4,5,6] # -> 1
452              
453             If subroutine make'nt, then died.
454

457             =head1 OPERATORS
458              
459             =head2 &{}
460              
461             It make the object is callable.
462              
463             my $PositiveInt = Aion::Type->new(
464             name => "PositiveInt",
465             test => sub { /^\d+$/ },
466             );
467            
468             local $_ = 10;
469             $PositiveInt->() # -> 1
470            
471             $_ = -1;
472             $PositiveInt->() # -> ""
473              
474             =head2 ""
475              
476             Stringify object.
477              
478             Aion::Type->new(name => "Int") . "" # => Int
479            
480             my $Enum = Aion::Type->new(name => "Enum", args => [qw/A B C/]);
481            
482             "$Enum" # => Enum['A', 'B', 'C']
483              
484             =head2 $a | $b
485              
486             It make new type as union of C<$a> and C<$b>.
487              
488             my $Int = Aion::Type->new(name => "Int", test => sub { /^-?\d+$/ });
489             my $Char = Aion::Type->new(name => "Char", test => sub { /^.\z/ });
490            
491             my $IntOrChar = $Int | $Char;
492            
493             77 ~~ $IntOrChar # => 1
494             "a" ~~ $IntOrChar # => 1
495             "ab" ~~ $IntOrChar # -> ""
496              
497             =head2 $a & $b
498              
499             It make new type as intersection of C<$a> and C<$b>.
500              
501             my $Int = Aion::Type->new(name => "Int", test => sub { /^-?\d+$/ });
502             my $Char = Aion::Type->new(name => "Char", test => sub { /^.\z/ });
503            
504             my $Digit = $Int & $Char;
505            
506             7 ~~ $Digit # => 1
507             77 ~~ $Digit # -> ""
508             "a" ~~ $Digit # -> ""
509              
510             =head2 ~ $a
511              
512             It make exclude type from C<$a>.
513              
514             my $Int = Aion::Type->new(name => "Int", test => sub { /^-?\d+$/ });
515            
516             "a" ~~ ~$Int; # => 1
517             5 ~~ ~$Int; # -> ""
